测试编写 ​

python
import pytest
from psorad.models import load_resnet50

class TestModelLoading:
    @pytest.fixture
    def model(self):
        return load_resnet50(pretrained=False)
    
    def test_model_forward(self, model):
        x = torch.randn(1, 3, 224, 224)
        y = model(x)
        assert y.shape == (1, 1000)
